Patient Journey & Clinical Operations Analytics
Track patient encounters, appointment compliance, clinical outcomes and operational throughput across care sites. Power staffing models, compliance reporting, and revenue cycle visibility.
Patient Encounters Revenue Cycle Readmission Flags HIPAA-aware Design SCD Type 2
Staging stg_patient_encounters — Raw EHR ingest with type casting ▼
Ingest raw encounter data from EHR. Generate MD5 surrogate keys, cast all types, and preserve source row for audit trail.
-- stg_patient_encounters.sql  (Staging / Bronze)
WITH source AS (
    SELECT * FROM {{ source('ehr_raw','encounters') }}
),
staged AS (
    SELECT
        MD5(encounter_id || patient_id)    AS encounter_key,
        encounter_id::VARCHAR              AS src_encounter_id,
        patient_id::VARCHAR               AS src_patient_id,
        provider_id::VARCHAR              AS src_provider_id,
        facility_code::VARCHAR            AS src_facility_code,
        encounter_date::DATE              AS encounter_date,
        admission_ts::TIMESTAMP_NTZ       AS admission_at,
        discharge_ts::TIMESTAMP_NTZ       AS discharge_at,
        UPPER(TRIM(encounter_status))     AS encounter_status_raw,
        billed_amount::NUMERIC(12,2)     AS billed_amount,
        insurance_code::VARCHAR           AS insurance_code,
        CURRENT_TIMESTAMP()               AS _loaded_at,
        _source_file
    FROM source
    WHERE encounter_id IS NOT NULL
      AND patient_id    IS NOT NULL
)
SELECT * FROM staged
Silver silver_encounters — Conformed with readmission flags & LOS ▼
Apply business rules, standardize status vocabulary, calculate length-of-stay, and flag 30-day readmissions via LAG window function.
-- silver_encounters.sql  (Silver)
WITH staged    AS ( SELECT * FROM {{ ref('stg_patient_encounters') }} ),
     facilities AS ( SELECT * FROM {{ ref('stg_facilities') }} )
SELECT
    e.encounter_key, e.src_encounter_id, e.src_patient_id,
    CASE e.encounter_status_raw
        WHEN 'COMP'  THEN 'Completed'
        WHEN 'CANC'  THEN 'Cancelled'
        WHEN 'NOSHO' THEN 'No-Show'
        WHEN 'SCHED' THEN 'Scheduled'
        ELSE 'Unknown'
    END                                          AS encounter_status,
    DATEDIFF('hour', e.admission_at,
        COALESCE(e.discharge_at, CURRENT_TIMESTAMP()))  AS los_hours,
    CASE WHEN
        LAG(e.encounter_date)
          OVER (PARTITION BY e.src_patient_id
                ORDER BY e.encounter_date)
        >= DATEADD('day', -30, e.encounter_date)
    THEN TRUE ELSE FALSE END                     AS is_readmission,
    f.facility_name, f.region,
    e.billed_amount, e.encounter_date, e._loaded_at
FROM staged e
LEFT JOIN facilities f ON e.src_facility_code = f.facility_code
WHERE e.billed_amount >= 0
  AND e.encounter_date >= '2020-01-01'
Gold fct_monthly_clinical_ops — KPI mart for BI reporting ▼
Aggregate to monthly grain: no-show rate, readmission rate, avg LOS, revenue per encounter — all windowed for prior-month LAG comparison.
-- fct_monthly_clinical_ops.sql  (Gold)
SELECT
    DATE_TRUNC('month', encounter_date)  AS report_month,
    region, facility_name,
    COUNT(*)                             AS total_encounters,
    COUNT(DISTINCT src_patient_id)       AS unique_patients,
    ROUND(SUM(CASE WHEN encounter_status = 'No-Show'
                THEN 1 ELSE 0 END) * 100.0
          / NULLIF(COUNT(*), 0), 2)    AS no_show_rate_pct,
    ROUND(SUM(is_readmission::INT) * 100.0
          / NULLIF(COUNT(*), 0), 2)    AS readmission_rate_pct,
    ROUND(AVG(los_hours), 1)           AS avg_los_hours,
    SUM(billed_amount)                  AS total_billed,
    LAG(COUNT(*)) OVER (
        PARTITION BY facility_name
        ORDER BY DATE_TRUNC('month', encounter_date)
    )                                   AS prev_month_encounters
FROM {{ ref('silver_encounters') }}
GROUP BY 1, 2, 3
ORDER BY report_month DESC, total_encounters DESC

Transaction Risk & Fraud Detection Analytics
Monitor transaction volumes, detect anomalies, calculate rolling risk scores, and surface suspicious account behavior for compliance and fraud operations teams.
Transaction Monitoring Rolling Aggregates Z-Score Anomaly Risk Scoring
Staging stg_transactions — Ledger ingest with ROW_NUMBER dedup ▼
Deduplicate on source transaction ID via ROW_NUMBER, normalize currency, and preserve full audit lineage.
-- stg_transactions.sql  (Staging / Bronze)
WITH source  AS ( SELECT * FROM {{ source('core_banking','transactions') }} ),
     deduped AS (
         SELECT *,
             ROW_NUMBER() OVER (
                 PARTITION BY transaction_id
                 ORDER BY processed_at DESC
             ) AS rn
         FROM source
     )
SELECT
    MD5(transaction_id)         AS txn_key,
    transaction_id::VARCHAR      AS src_transaction_id,
    account_id::VARCHAR          AS src_account_id,
    merchant_id::VARCHAR         AS src_merchant_id,
    processed_at::TIMESTAMP_NTZ AS processed_at,
    transaction_date::DATE       AS txn_date,
    amount::NUMERIC(14,2)       AS txn_amount_local,
    currency_code::VARCHAR(3)   AS currency_code,
    UPPER(TRIM(txn_type))       AS txn_type_raw,
    CURRENT_TIMESTAMP()         AS _loaded_at
FROM deduped
WHERE rn = 1 AND amount > 0 AND transaction_id IS NOT NULL
Silver silver_transactions — Rolling 7d spend & z-score risk flags ▼
Compute rolling 7-day spend, daily velocity counts, z-score anomaly signals, and business rule risk flags per account.
-- silver_transactions.sql  (Silver)
WITH txns AS ( SELECT * FROM {{ ref('stg_transactions') }} ),
rolling AS (
    SELECT *,
        SUM(txn_amount_local) OVER (
            PARTITION BY src_account_id ORDER BY txn_date
            ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
        )                              AS rolling_7d_spend,
        COUNT(*) OVER (
            PARTITION BY src_account_id, txn_date
        )                              AS daily_txn_count,
        (txn_amount_local - AVG(txn_amount_local) OVER
            (PARTITION BY src_account_id))
        / NULLIF(STDDEV(txn_amount_local) OVER
            (PARTITION BY src_account_id), 0)  AS amount_z_score
    FROM txns
)
SELECT r.*,
    CASE
        WHEN r.rolling_7d_spend > 10000  THEN TRUE
        WHEN r.daily_txn_count  > 20     THEN TRUE
        WHEN r.amount_z_score   > 3      THEN TRUE
        ELSE FALSE
    END                                AS is_high_risk,
    CASE r.txn_type_raw
        WHEN 'PUR' THEN 'Purchase'
        WHEN 'REF' THEN 'Refund'
        WHEN 'WD'  THEN 'Withdrawal'
        WHEN 'TRF' THEN 'Transfer'
        ELSE 'Other'
    END                                AS txn_type
FROM rolling r
Gold fct_account_risk_summary — Daily risk mart for fraud ops ▼
Daily per-account risk summary: flagged transaction count, risk rate %, total flagged exposure, and velocity score for fraud analyst dashboards.
-- fct_account_risk_summary.sql  (Gold)
SELECT
    txn_date, src_account_id, currency_code,
    COUNT(*)                              AS total_transactions,
    SUM(txn_amount_local)                AS total_spend,
    SUM(is_high_risk::INT)               AS flagged_txn_count,
    ROUND(SUM(is_high_risk::INT) * 100.0
          / NULLIF(COUNT(*), 0), 2)     AS risk_rate_pct,
    SUM(CASE WHEN is_high_risk
            THEN txn_amount_local ELSE 0 END) AS flagged_exposure,
    ROUND(COUNT(*) / 24.0, 2)           AS txn_velocity_per_hr,
    SUM(txn_amount_local)
      - LAG(SUM(txn_amount_local), 30)
        OVER (PARTITION BY src_account_id
              ORDER BY txn_date)           AS spend_30d_delta
FROM {{ ref('silver_transactions') }}
GROUP BY 1, 2, 3
ORDER BY risk_rate_pct DESC

Customer 360 & RFM Segmentation
Unify POS and e-commerce orders into customer lifetime value models, RFM segmentation, and inventory-linked sales analytics for merchandising and retention teams.
Multi-Source Union Customer LTV RFM Segmentation NTILE Scoring
Staging stg_orders — POS + e-commerce UNION with channel tag ▼
UNION ALL POS and Shopify order sources, tag each row with channel origin, normalize discount handling and cast financial fields.
-- stg_orders.sql  (Staging / Bronze)
WITH
pos_src  AS (SELECT *, 'POS'  AS order_channel FROM {{ source('pos_system','sales') }}),
ecom_src AS (SELECT *, 'ECOM' AS order_channel FROM {{ source('shopify','orders') }}),
unioned  AS (SELECT * FROM pos_src UNION ALL SELECT * FROM ecom_src)
SELECT
    MD5(order_id || order_channel)         AS order_key,
    order_id::VARCHAR                       AS src_order_id,
    customer_id::VARCHAR                    AS src_customer_id,
    order_channel,
    order_date::DATE                        AS order_date,
    gross_amount::NUMERIC(12,2)           AS gross_amount,
    discount_amount::NUMERIC(10,2)        AS discount_amount,
    gross_amount - COALESCE(discount_amount, 0)  AS net_amount,
    store_id::VARCHAR                       AS store_id,
    UPPER(payment_method)                   AS payment_method,
    CURRENT_TIMESTAMP()                     AS _loaded_at
FROM unioned
WHERE order_id    IS NOT NULL
  AND customer_id IS NOT NULL
Silver silver_customer_orders — RFM signals via window functions ▼
Compute Recency (days since last order), Frequency (order rank), Monetary (cumulative LTV), and first-order flag per customer.
-- silver_customer_orders.sql  (Silver)
WITH orders AS (SELECT * FROM {{ ref('stg_orders') }})
SELECT *,
    -- Recency
    DATEDIFF('day', order_date,
        MAX(order_date) OVER (PARTITION BY src_customer_id)
    )                                      AS days_since_last_order,
    -- Frequency
    ROW_NUMBER() OVER (
        PARTITION BY src_customer_id
        ORDER BY order_date
    )                                      AS customer_order_rank,
    -- Monetary: cumulative LTV
    SUM(net_amount) OVER (
        PARTITION BY src_customer_id
        ORDER BY order_date
        ROWS UNBOUNDED PRECEDING
    )                                      AS cumulative_ltv,
    CASE WHEN ROW_NUMBER() OVER (
        PARTITION BY src_customer_id ORDER BY order_date
    ) = 1 THEN TRUE ELSE FALSE END       AS is_first_order
FROM orders
Gold fct_customer_segments — NTILE RFM scoring & segment labels ▼
Score each RFM dimension with NTILE(5), compute composite score, and classify into Champion / Loyal / Potential / At-Risk / Lost segments.
-- fct_customer_segments.sql  (Gold)
WITH summary AS (
    SELECT src_customer_id,
        MAX(order_date) AS last_order_date,
        COUNT(*)        AS total_orders,
        SUM(net_amount) AS total_ltv,
        MIN(days_since_last_order) AS recency_days
    FROM {{ ref('silver_customer_orders') }} GROUP BY 1
),
scored AS (
    SELECT *,
        NTILE(5) OVER (ORDER BY recency_days ASC) AS r_score,
        NTILE(5) OVER (ORDER BY total_orders  ASC) AS f_score,
        NTILE(5) OVER (ORDER BY total_ltv     ASC) AS m_score
    FROM summary
)
SELECT *,
    (r_score + f_score + m_score) / 3.0    AS rfm_composite,
    CASE
        WHEN r_score >= 4 AND f_score >= 4  THEN 'Champion'
        WHEN f_score >= 3 AND m_score >= 3  THEN 'Loyal'
        WHEN r_score <= 2 AND f_score >= 3  THEN 'At-Risk'
        WHEN r_score = 1                      THEN 'Lost'
        ELSE 'Potential'
    END                                     AS customer_segment
FROM scored ORDER BY rfm_composite DESC

Equipment OEE & Production Downtime Analytics
Calculate Overall Equipment Effectiveness (OEE), track unplanned downtime root causes, and identify production line bottlenecks from IoT sensor and MES event data.
OEE Calculation Shift-Level Grain IoT / MES Integration 7-Day Rolling Avg
Staging stg_production_events — MES stream with shift tagging ▼
Ingest high-frequency MES events, parse event types, assign shift names from event hour CASE logic, and cast timestamps to sub-second precision.
-- stg_production_events.sql  (Staging / Bronze)
WITH source AS (SELECT * FROM {{ source('mes_system','production_events') }})
SELECT
    MD5(event_id)                        AS event_key,
    event_id::VARCHAR                    AS src_event_id,
    machine_id::VARCHAR                  AS src_machine_id,
    line_id::VARCHAR                     AS src_line_id,
    event_ts::TIMESTAMP_NTZ             AS event_at,
    event_ts::DATE                      AS event_date,
    CASE
        WHEN HOUR(event_ts) BETWEEN 6  AND 13 THEN 'Day'
        WHEN HOUR(event_ts) BETWEEN 14 AND 21 THEN 'Evening'
        ELSE 'Night'
    END                                  AS shift_name,
    UPPER(TRIM(event_type))              AS event_type_raw,
    units_produced::INT                  AS units_produced,
    units_rejected::INT                  AS units_rejected,
    downtime_minutes::NUMERIC(8,2)      AS downtime_minutes,
    downtime_reason_code::VARCHAR        AS downtime_reason_code,
    CURRENT_TIMESTAMP()                 AS _loaded_at
FROM source
WHERE event_id IS NOT NULL AND machine_id IS NOT NULL
Silver silver_machine_performance — OEE Availability & Quality ▼
Aggregate to machine × shift grain, compute planned production minutes, and calculate OEE Availability and Quality components.
-- silver_machine_performance.sql  (Silver)
WITH
events    AS (SELECT * FROM {{ ref('stg_production_events') }}),
shift_agg AS (
    SELECT src_machine_id, src_line_id, event_date, shift_name,
        SUM(units_produced)   AS total_produced,
        SUM(units_rejected)   AS total_rejected,
        SUM(downtime_minutes) AS total_downtime_min,
        480                   AS planned_minutes  -- 8-hr shift
    FROM events
    GROUP BY 1,2,3,4
),
oee_calc AS (
    SELECT *,
        (planned_minutes - total_downtime_min)
          / NULLIF(planned_minutes, 0) * 100  AS availability_pct,
        (total_produced - total_rejected)
          / NULLIF(total_produced, 0) * 100   AS quality_pct
    FROM shift_agg
)
SELECT *,
    ROUND(availability_pct * quality_pct / 100, 2) AS oee_score
FROM oee_calc
Gold fct_line_oee_daily — Production KPIs with 7-day rolling avg ▼
Roll up to daily line-level OEE. 7-day rolling average for trend tracking. FIRST_VALUE identifies the highest-downtime machine each day.
-- fct_line_oee_daily.sql  (Gold)
SELECT
    event_date, src_line_id,
    AVG(oee_score)                           AS avg_line_oee,
    MIN(oee_score)                           AS worst_machine_oee,
    AVG(AVG(oee_score)) OVER (
        PARTITION BY src_line_id
        ORDER BY event_date
        ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
    )                                        AS oee_7d_rolling,
    SUM(total_produced)                     AS daily_units_produced,
    SUM(total_rejected)                     AS daily_units_rejected,
    SUM(total_downtime_min)                 AS total_downtime_min,
    FIRST_VALUE(src_machine_id) OVER (
        PARTITION BY src_line_id, event_date
        ORDER BY total_downtime_min DESC
    )                                        AS highest_downtime_machine
FROM {{ ref('silver_machine_performance') }}
GROUP BY event_date, src_line_id, src_machine_id
ORDER BY event_date DESC, avg_line_oee ASC

Staging stg_employees — HRIS ingest with SHA2 PII hashing ▼
Ingest employee master records from ADP. Hash PII fields with SHA2-256 for safe downstream use, normalize employment status codes.
-- stg_employees.sql  (Staging / Bronze)
WITH source AS (SELECT * FROM {{ source('adp','employee_master') }})
SELECT
    MD5(employee_id)                  AS employee_key,
    employee_id::VARCHAR              AS src_employee_id,
    SHA2(email_address, 256)         AS email_hash,  -- PII-safe
    department_code::VARCHAR          AS department_code,
    job_title::VARCHAR                AS job_title,
    manager_id::VARCHAR               AS manager_id,
    cost_center::VARCHAR              AS cost_center,
    location_code::VARCHAR            AS location_code,
    hire_date::DATE                   AS hire_date,
    termination_date::DATE            AS termination_date,
    UPPER(TRIM(employment_status))   AS employment_status_raw,
    salary_band::VARCHAR              AS salary_band,
    annual_salary::NUMERIC(12,2)    AS annual_salary,
    CURRENT_TIMESTAMP()              AS _loaded_at
FROM source
WHERE employee_id IS NOT NULL
Silver silver_workforce — Tenure bands, active flags, status vocab ▼
Compute tenure months, apply tenure banding, flag active employees, and standardize employment status to Voluntary / Involuntary / Active.
-- silver_workforce.sql  (Silver)
WITH employees AS (SELECT * FROM {{ ref('stg_employees') }})
SELECT
    employee_key, src_employee_id, department_code,
    job_title, cost_center, salary_band, annual_salary,
    hire_date, termination_date,
    CASE WHEN termination_date IS NULL
          OR  termination_date > CURRENT_DATE()
        THEN TRUE ELSE FALSE END                  AS is_active,
    DATEDIFF('month', hire_date,
        COALESCE(termination_date, CURRENT_DATE()))  AS tenure_months,
    CASE
        WHEN DATEDIFF('month',hire_date,CURRENT_DATE()) <  12 THEN '0-1 yr'
        WHEN DATEDIFF('month',hire_date,CURRENT_DATE()) <  36 THEN '1-3 yrs'
        WHEN DATEDIFF('month',hire_date,CURRENT_DATE()) <  60 THEN '3-5 yrs'
        ELSE '5+ yrs'
    END                                             AS tenure_band,
    CASE employment_status_raw
        WHEN 'TERM-VOL'   THEN 'Voluntary'
        WHEN 'TERM-INVOL' THEN 'Involuntary'
        WHEN 'ACTIVE'    THEN 'Active'
        ELSE 'Other'
    END                                             AS employment_status
FROM employees
Gold fct_monthly_attrition — Workforce KPI mart with LAG MoM ▼
Monthly attrition rate, active headcount, and voluntary separation rate per department and tenure band — with LAG for month-over-month comparison.
-- fct_monthly_attrition.sql  (Gold)
SELECT
    DATE_TRUNC('month', hire_date)    AS cohort_month,
    department_code, salary_band, tenure_band,
    COUNT(*)                           AS total_headcount,
    SUM(is_active::INT)               AS active_headcount,
    SUM(CASE WHEN NOT is_active THEN 1 ELSE 0 END)
                                       AS separated_headcount,
    ROUND(SUM(CASE WHEN NOT is_active THEN 1 ELSE 0 END) * 100.0
          / NULLIF(COUNT(*), 0), 2)  AS attrition_rate_pct,
    ROUND(SUM(CASE WHEN employment_status = 'Voluntary'
                THEN 1 ELSE 0 END) * 100.0
          / NULLIF(COUNT(*), 0), 2)  AS voluntary_sep_rate_pct,
    ROUND(AVG(tenure_months), 1)    AS avg_tenure_months,
    ROUND(AVG(annual_salary), 0)   AS avg_annual_salary,
    LAG(SUM(is_active::INT)) OVER (
        PARTITION BY department_code
        ORDER BY DATE_TRUNC('month', hire_date)
    )                                  AS prev_month_headcount
FROM {{ ref('silver_workforce') }}
GROUP BY 1,2,3,4
ORDER BY cohort_month DESC, attrition_rate_pct DESC
